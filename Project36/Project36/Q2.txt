#include<stdio.h>

int count(char *c)
{
	int i, count;
	count = 0;

	for (i = 0; c[i] != NULL; i++)
	{
		if (c[i] >= 'a'&&c[i] <= 'z' || c[i] >= 'A'&& c[i] <= 'Z' || c[i] == ' ')
			continue;
		else //한글
			count++;
	}
	return count / 2;
}
